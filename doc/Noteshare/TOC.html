def initialize(document)
  @document = document
  @toc_array = @document.toc

  @table = []

  @toc_array.each do |item|
    toc_item =  TOCItem.from_hash(item)
    @table << toc_item
  end
end

def change_title(id, new_title)
  target = nil
  table.each do |item|
    if item.id == id
      target = item
      target.title = new_title
      break
    end
  end
  target
end

def display
  str = "\n-------\nTABLE:\n"
  @table.each do |item|
    str << item.display
  end
  str << "-------\n#{@table.count}\n"
end
