def hash_value(arg = {})
  key_value_separator = arg[:key_value_separator] || ':'
  item_separator = arg[:item_separator] || ','
  items = self.split(item_separator)
  items = items.map{ |item| item.strip }
  hash = {}
  items.each do |item|
    a, b = item.split(key_value_separator)
    a = a.strip
    b = b.strip if b
    b ||= ''
    hash[a] = b
  end
  hash
end
