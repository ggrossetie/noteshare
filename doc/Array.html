def head(k)
  return [] if self == []
  n = self.length
  return self if k >=  n
  self[0, k+1]
end

def random_indices(number_of_indices)
  output = []
  count = 0
  while output.count < number_of_indices and count < 2*number_of_indices do
    j = SecureRandom.random_number(self.count)
    if !output.include? j
      output << j
    end
    count += 1
  end
  return output
end

def random_sublist(size)

  output = []
  indices = random_indices(size)
  indices.each do |index|
    output << self[index]
  end
  return output

end

def tail(k)
  return [] if self == []
  n = self.length
  return self if k >= n
  self[k+1, n - k-1]
end
