def get_ad_element1(text, tag_prefix, tag_suffix)

  # if Rails.env =='development' or Rails.env =='test'
  end_string = "\r\n--\r\n(.*?)\r\n--\r\n" # ENV[END_STRING]

def get_html_element(text, tag)

  match = re.match text
  match ? match[1] : ''
end
